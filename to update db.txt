-- Step 1: Add missing columns to licenses table
ALTER TABLE licenses 
    ADD COLUMN IF NOT EXISTS license_id UUID DEFAULT gen_random_uuid(),
    ADD COLUMN IF NOT EXISTS license_key VARCHAR(255),
    ADD COLUMN IF NOT EXISTS customer_name VARCHAR(255),
    ADD COLUMN IF NOT EXISTS customer_email VARCHAR(255),
    ADD COLUMN IF NOT EXISTS subscription_type VARCHAR(50) DEFAULT 'yearly',
    ADD COLUMN IF NOT EXISTS status VARCHAR(50) DEFAULT 'active',
    ADD COLUMN IF NOT EXISTS start_date DATE DEFAULT CURRENT_DATE,
    ADD COLUMN IF NOT EXISTS expiry_date DATE DEFAULT (CURRENT_DATE + INTERVAL '1 year'),
    ADD COLUMN IF NOT EXISTS max_devices INTEGER DEFAULT 1,
    ADD COLUMN IF NOT EXISTS created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP;

-- Populate license_id for existing rows
UPDATE licenses SET license_id = gen_random_uuid() WHERE license_id IS NULL;

-- Add unique constraint on license_key if it doesn't exist (optional, for license key lookups)
DO $$ 
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM information_schema.table_constraints 
        WHERE table_name = 'licenses' 
        AND constraint_name = 'licenses_license_key_key'
    ) THEN
        -- Only add if no duplicate license_keys exist
        IF NOT EXISTS (
            SELECT license_key FROM licenses 
            WHERE license_key IS NOT NULL
            GROUP BY license_key HAVING COUNT(*) > 1
        ) THEN
            ALTER TABLE licenses ADD CONSTRAINT licenses_license_key_key UNIQUE (license_key);
        END IF;
    END IF;
END $$;

-- Step 2: Create device_activations table
CREATE TABLE IF NOT EXISTS device_activations (
    activation_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    store_id UUID NOT NULL REFERENCES licenses(store_id) ON DELETE CASCADE,
    license_id UUID,
    device_fingerprint VARCHAR(255) NOT NULL,
    device_name VARCHAR(255),
    device_info JSONB,
    activated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    last_validated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_active BOOLEAN DEFAULT true
);

-- Add unique constraint on store_id + device_fingerprint
DO $$ 
BEGIN
    IF NOT EXISTS (
        SELECT 1 FROM information_schema.table_constraints 
        WHERE table_name = 'device_activations' 
        AND constraint_name = 'device_activations_store_device_unique'
    ) THEN
        ALTER TABLE device_activations 
            ADD CONSTRAINT device_activations_store_device_unique 
            UNIQUE (store_id, device_fingerprint);
    END IF;
END $$;

-- Step 3: Create license_validations table
CREATE TABLE IF NOT EXISTS license_validations (
    validation_id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    store_id UUID REFERENCES licenses(store_id),
    license_id UUID,
    device_fingerprint VARCHAR(255),
    validation_result VARCHAR(50),
    validation_date TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    ip_address VARCHAR(45),
    user_agent TEXT
);

-- Step 4: Create indexes
CREATE INDEX IF NOT EXISTS idx_licenses_store_id ON licenses(store_id);
CREATE INDEX IF NOT EXISTS idx_licenses_license_key ON licenses(license_key) WHERE license_key IS NOT NULL;
CREATE INDEX IF NOT EXISTS idx_licenses_status ON licenses(status);
CREATE INDEX IF NOT EXISTS idx_licenses_valid ON licenses(valid);
CREATE INDEX IF NOT EXISTS idx_device_activations_store_id ON device_activations(store_id);
CREATE INDEX IF NOT EXISTS idx_device_activations_license_id ON device_activations(license_id);
CREATE INDEX IF NOT EXISTS idx_device_activations_fingerprint ON device_activations(device_fingerprint);
CREATE INDEX IF NOT EXISTS idx_device_activations_active ON device_activations(is_active);
CREATE INDEX IF NOT EXISTS idx_license_validations_store_id ON license_validations(store_id);
CREATE INDEX IF NOT EXISTS idx_license_validations_date ON license_validations(validation_date);
